import requests
import os

# Configuration
STORAGE_NODE_URL = os.getenv("CSOP_STORAGE_URL", "https://your-storage-node.vercel.app")

class CSOPClient:
    def __init__(self, base_url):
        self.base_url = base_url
    
    def write(self, collection, key, data):
        """√âcrire des donn√©es"""
        response = requests.post(
            f"{self.base_url}?action=write",
            json={
                "collection": collection,
                "key": key,
                "data": data
            }
        )
        return response.json()
    
    def read(self, collection, key):
        """Lire des donn√©es"""
        response = requests.get(
            f"{self.base_url}?action=read&collection={collection}&key={key}"
        )
        return response.json()
    
    def list(self, collection):
        """Lister tous les documents"""
        response = requests.get(
            f"{self.base_url}?action=list&collection={collection}"
        )
        return response.json()
    
    def update(self, collection, key, data):
        """Mettre √† jour"""
        response = requests.put(
            f"{self.base_url}?action=update",
            json={
                "collection": collection,
                "key": key,
                "data": data
            }
        )
        return response.json()
    
    def delete(self, collection, key):
        """Supprimer"""
        response = requests.delete(
            f"{self.base_url}?action=delete&collection={collection}&key={key}"
        )
        return response.json()

# Utilisation
db = CSOPClient(STORAGE_NODE_URL)

# √âcrire
db.write("users", "user_123", {"name": "Anzize", "role": "CEO"})

# Lire
user = db.read("users", "user_123")
print(user)

# Lister
all_users = db.list("users")
print(f"Total users: {len(all_users['data'])}")
